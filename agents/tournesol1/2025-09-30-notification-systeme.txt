  TASKS_FILE = 'tasks.json'

  def create_task(title, description):

          tasks = json.load(f)
          for task in tasks:
              if task['title'] == title:
                  task['description'] = description
                  task['status'] = status
                  f.seek(0)
                  json.dump(tasks, f)
                  f.truncate()
                  break

  # Fonction pour supprimer une tâche
  def delete_task(title):
      with open(TASKS_FILE, 'r+') as f:
          tasks = json.load(f)

                  f.seek(0)
                  json.dump(tasks, f)
                  f.truncate()

  # Création de l'interface CLI
  parser = argparse.ArgumentParser(description='TODO list application')

  ```python
  import argparse
  import json
  import os

  # Définition des constantes
  TASKS_FILE = 'tasks.json'

  # Fonction pour créer une tâche
  def create_task(title, description, status):
      task = {
          'title': title,
          'description': description,
          'status': status
      }
      with open(TASKS_FILE, 'r+') as f:
          try:
              tasks = json.load(f)
          except ValueError:
              tasks = []
          tasks.append(task)
          f.seek(0)
          json.dump(tasks, f)
          f.truncate()

  # Fonction pour lire les tâches
  def read_tasks():
      with open(TASKS_FILE, 'r') as f:
          try:
              tasks = json.load(f)
              for task in tasks:
                  print(f"Title: {task['title']}")
                  print(f"Description: {task['description']}")
                  print(f"Status: {task['status']}")
                  print()
          except FileNotFoundError:
              print("Aucune tâche enregistrée")

  # Fonction pour mettre à jour une tâche
  def update_task(title, description, status):
      with open(TASKS_FILE, 'r+') as f:
          try:
              tasks = json.load(f)
              for task in tasks:
                  if task['title'] == title:
                      task['description'] = description
                      task['status'] = status
                      f.seek(0)
                      json.dump(tasks, f)
                      f.truncate()
                      break
              else:
                  print("Tâche non trouvée")
          except FileNotFoundError:
              print("Aucune tâche enregistrée")

  # Fonction pour supprimer une tâche
  def delete_task(title):
      with open(TASKS_FILE, 'r+') as f:
          try:
              tasks = json.load(f)
              for task in tasks:
                  if task['title'] == title:
                      tasks.remove(task)
                      f.seek(0)
                      json.dump(tasks, f)
                      f.truncate()
                      break
              else:
                  print("Tâche non trouvée")
          except FileNotFoundError:
              print("Aucune tâche enregistrée")

  # Création de l'interface CLI
  parser = argparse.ArgumentParser(description='TODO list application')
  subparsers = parser.add_subparsers(dest='command')

  create_parser = subparsers.add_parser('create')
  create_parser.add_argument('--title', required=True)
  create_parser.add_argument('--description', required=True)
  create_parser.add_argument('--status', required=True)

  read_parser = subparsers.add_parser('read')

  update_parser = subparsers.add_parser('update')
  update_parser.add_argument('--title', required=True)
  update_parser.add_argument('--description', required=True)
  update_parser.add_argument('--status', required=True)

  delete_parser = subparsers.add_parser('delete')
  delete_parser.add_argument('--title', required=True)

  args = parser.parse_args()

  if args.command == 'create':
      create_task(args.title, args.description, args.status)
  elif args.command == 'read':
      read_tasks()
  elif args.command == 'update':
      update_task(args.title, args.description, args.status)
  elif args.command == 'delete':
      delete_task(args.title)
  ```
